Introduction

This report mirrors qc_rRNA.html but evaluates an alternative site-calling strategy based on a per-site contingency-table (chi-squared) test rather than simple ratio/depth thresholds.

For every uridine (T) position we build a 2x2 table comparing the treated and (pooled) untreated samples:

Conversion (C) No conversion (T)
Treated a b
Untreated c d

and test the null hypothesis that the conversion rate is the same in both samples (\(X^2 = \frac{n(ad-bc)^2}{(a+b)(c+d)(a+c)(b+d)}\), 1 d.f.). A site is called pseudouridylated when:

  1. the chi-squared Benjamini-Hochberg FDR < cutoff, and
  2. the conversion-rate difference (treated − untreated) > cutoff.

We ran this strategy on the same four cell lines: CL18870, CL18909, CL18916, and CL19099 (four treated replicates each).

We compare our results against two published datasets:

  • BACS — PMID 39349603
  • PRAISE — PMID 36997645

Examining replicates for each cell line

For each cell line we examine how consistently the chi-squared site-calling strategy identifies pseudouridylated rRNA positions across its four treated replicates, using the per-replicate filtered chi-squared output (out_chisq/filter.<sample>.rRNA.pseusite.txt.gz), the same per-site table that feeds into the pooling step described below. Each row of this table already carries the chi-squared statistics (pval, fdr, rate_diff) alongside the treated conversion ratio (ratio) for every candidate site.

Because “detection” under this strategy is a statistical call rather than a simple ratio/depth threshold, we repeat each check at four cutoff combinations spanning two stringency dimensions:

Combination Max FDR Min rate difference
1 < 0.001 > 0.05
2 < 0.001 > 0.10
3 < 0.01 > 0.05
4 < 0.01 > 0.10

For each combination we run four checks:

  1. Site overlap (Jaccard index) — how many sites called significant are shared between each pair of replicates, expressed as a set-similarity score.
  2. %pU distribution — histogram and density of the (treated) conversion ratio at sites called significant in each replicate, used to judge overall signal strength and consistency.
  3. Pairwise Pearson correlation — quantitative agreement of %pU at sites called significant in both replicates of a pair.
  4. Pairwise Pearson heatmap — compact summary of all pairwise correlations.

# ── Data loading ──────────────────────────────────────────────────────────────
read_rRNA_chisq_replicate <- function(cl, sample_id, chisq_dir) {
  f <- file.path(chisq_dir, sprintf("filter.%s.rRNA.pseusite.txt.gz", sample_id))
  if (!file.exists(f)) {
    warning("File not found: ", f)
    return(NULL)
  }
  dt <- fread(cmd = paste("zcat", f), header = TRUE, sep = "\t",
              colClasses = list(character = "ref"), data.table = FALSE)
  dt %>%
    mutate(site = paste(ref, pos, strand, sep = ":")) %>%
    select(site, ratio, fdr, rate_diff)
}

sites_passing <- function(dt, combo) dt$site[dt$fdr < combo$fdr & dt$rate_diff > combo$diff]

shared_passing_pair <- function(rep_tabs, p, combo) {
  d1 <- rep_tabs[[p[1]]]; d2 <- rep_tabs[[p[2]]]
  s1 <- sites_passing(d1, combo); s2 <- sites_passing(d2, combo)
  shared <- intersect(s1, s2)
  if (length(shared) < 5) return(NULL)
  merge(d1[d1$site %in% shared, c("site", "ratio")],
        d2[d2$site %in% shared, c("site", "ratio")],
        by = "site", suffixes = c("_1", "_2"))
}

# ── Distribution plots ────────────────────────────────────────────────────────
plot_distribution_chisq <- function(rep_tabs, rep_names, cl, combo) {
  lapply(rep_names, function(samp) {
    dt <- rep_tabs[[samp]]
    df <- dt[dt$fdr < combo$fdr & dt$rate_diff > combo$diff, , drop = FALSE]
    if (nrow(df) == 0) return(NULL)

    med_val  <- median(df$ratio)
    mean_val <- mean(df$ratio)

    ggplot(df, aes(x = ratio)) +
      geom_histogram(aes(y = after_stat(density)),
                     binwidth = 0.02, fill = "steelblue",
                     color = "white", linewidth = 0.2, alpha = 0.7) +
      geom_density(color = "navy", linewidth = 0.7) +
      geom_vline(xintercept = med_val,  color = "red",
                 linetype = "dashed",  linewidth = 0.6) +
      geom_vline(xintercept = mean_val, color = "darkorange",
                 linetype = "dotted",  linewidth = 0.6) +
      annotate("text", x = Inf, y = Inf,
               label = sprintf("n = %d\nmedian = %.3f\nmean = %.3f",
                               nrow(df), med_val, mean_val),
               hjust = 1.05, vjust = 1.05, size = 3, family = "mono") +
      scale_x_continuous(limits = c(0, 1), breaks = seq(0, 1, 0.1)) +
      labs(
        title    = samp,
        subtitle = combo$label,
        x        = "%pU  (C / C+T, treated)",
        y        = "Density"
      ) +
      theme_bw(base_size = 10) +
      theme(axis.text.x = element_text(angle = 45, hjust = 1))
  })
}

# ── Scatter plots ─────────────────────────────────────────────────────────────
make_scatter_plots_chisq <- function(rep_tabs, pairs, cl, combo) {
  lapply(pairs, function(p) {
    m <- shared_passing_pair(rep_tabs, p, combo)
    if (is.null(m)) return(NULL)

    r <- cor(m$ratio_1, m$ratio_2, method = "pearson")
    ggplot(m, aes(x = ratio_1, y = ratio_2)) +
      geom_point(alpha = 0.4, size = 0.8) +
      geom_abline(slope = 1, intercept = 0,
                  color = "red", linetype = "dashed") +
      annotate("text", x = Inf, y = -Inf,
               label = sprintf("r = %.3f\nn = %d", r, nrow(m)),
               hjust = 1.1, vjust = -0.3, size = 3.5) +
      labs(
        title = sprintf("%s vs %s\n[%s]", p[1], p[2], combo$label),
        x     = sprintf("%%pU  %s", p[1]),
        y     = sprintf("%%pU  %s", p[2])
      ) +
      theme_bw(base_size = 10)
  })
}

# ── Pearson correlation heatmap ───────────────────────────────────────────────
make_cor_heatmap_chisq <- function(rep_tabs, pairs, rep_names, cl, combo) {
  cor_df <- do.call(rbind, lapply(pairs, function(p) {
    m <- shared_passing_pair(rep_tabs, p, combo)
    r <- if (!is.null(m)) round(cor(m$ratio_1, m$ratio_2, method = "pearson"), 4) else NA_real_
    n <- if (!is.null(m)) nrow(m) else 0L
    data.frame(rep1 = p[1], rep2 = p[2], r = r, n = n, stringsAsFactors = FALSE)
  }))

  mat <- matrix(NA_real_, length(rep_names), length(rep_names),
                dimnames = list(rep_names, rep_names))
  diag(mat) <- 1
  n_mat <- matrix(NA_integer_, length(rep_names), length(rep_names),
                  dimnames = list(rep_names, rep_names))
  for (i in seq_len(nrow(cor_df))) {
    mat[cor_df$rep1[i], cor_df$rep2[i]] <- cor_df$r[i]
    mat[cor_df$rep2[i], cor_df$rep1[i]] <- cor_df$r[i]
    n_mat[cor_df$rep1[i], cor_df$rep2[i]] <- cor_df$n[i]
    n_mat[cor_df$rep2[i], cor_df$rep1[i]] <- cor_df$n[i]
  }

  short <- function(s) sub("^[^_]+_", "", s)
  lev   <- short(rep_names)

  plot_df <- expand.grid(x = rep_names, y = rep_names,
                         stringsAsFactors = FALSE) |>
    rowwise() |>
    mutate(
      r     = mat[x, y],
      n     = n_mat[x, y],
      label = case_when(
        x == y   ~ "1.000",
        is.na(r) ~ "NA",
        TRUE     ~ sprintf("%.3f\n(n=%d)", r, n)
      )
    ) |>
    ungroup() |>
    mutate(
      x_lab = factor(short(x), levels = lev),
      y_lab = factor(short(y), levels = rev(lev))
    )

  ggplot(plot_df, aes(x = x_lab, y = y_lab, fill = r)) +
    geom_tile(color = "white", linewidth = 0.8) +
    geom_text(aes(label = label), size = 3, lineheight = 0.9) +
    scale_fill_gradient2(
      low  = "#1a9641", mid = "#ffffbf", high = "#d7191c",
      midpoint = 0.5, limits = c(0, 1), na.value = "grey90",
      name = "Pearson r"
    ) +
    labs(title = cl, subtitle = combo$label, x = NULL, y = NULL) +
    coord_fixed() +
    theme_bw(base_size = 10) +
    theme(
      axis.text.x       = element_text(angle = 35, hjust = 1),
      panel.grid        = element_blank(),
      panel.border      = element_blank(),
      legend.key.height = unit(0.8, "cm")
    )
}

# ── Jaccard index helpers ─────────────────────────────────────────────────────
jaccard_index <- function(a, b) {
  n_int <- length(intersect(a, b))
  n_uni <- length(union(a, b))
  if (n_uni == 0) return(NA_real_)
  n_int / n_uni
}

make_jaccard_heatmap_chisq <- function(rep_tabs, pairs, rep_names, cl, combo) {
  site_sets <- lapply(rep_tabs, sites_passing, combo = combo)

  mat <- matrix(NA_real_, length(rep_names), length(rep_names),
                dimnames = list(rep_names, rep_names))
  diag(mat) <- 1
  n_int_mat <- matrix(NA_integer_, length(rep_names), length(rep_names),
                      dimnames = list(rep_names, rep_names))
  for (p in pairs) {
    sa <- site_sets[[p[1]]]; sb <- site_sets[[p[2]]]
    j   <- jaccard_index(sa, sb)
    n_i <- length(intersect(sa, sb))
    mat[p[1], p[2]]       <- j;   mat[p[2], p[1]]       <- j
    n_int_mat[p[1], p[2]] <- n_i; n_int_mat[p[2], p[1]] <- n_i
  }

  short <- function(s) sub("^[^_]+_", "", s)
  lev   <- short(rep_names)

  plot_df <- expand.grid(x = rep_names, y = rep_names,
                         stringsAsFactors = FALSE) |>
    rowwise() |>
    mutate(
      j     = mat[x, y],
      n_int = n_int_mat[x, y],
      label = case_when(
        x == y   ~ "1.000",
        is.na(j) ~ "NA",
        TRUE     ~ sprintf("%.3f\n(n=%d)", j, n_int)
      )
    ) |>
    ungroup() |>
    mutate(
      x_lab = factor(short(x), levels = lev),
      y_lab = factor(short(y), levels = rev(lev))
    )

  ggplot(plot_df, aes(x = x_lab, y = y_lab, fill = j)) +
    geom_tile(color = "white", linewidth = 0.8) +
    geom_text(aes(label = label), size = 3, lineheight = 0.9) +
    scale_fill_gradient2(
      low      = "#1a9641", mid = "#ffffbf", high = "#d7191c",
      midpoint = 0.5, limits = c(0, 1), na.value = "grey90",
      name     = "Jaccard"
    ) +

    )
}

Site overlap (Jaccard index)

The Jaccard index measures set similarity between the sets of sites called significant in two replicates at a given cutoff combination:

\[J(A, B) = \frac{|A \cap B|}{|A \cup B|}\]

A score of 1 means every called site appears in both replicates; 0 means no shared sites. The four panels correspond to the four cutoff combinations above (stricter combinations are more conservative and typically show fewer, but more reproducible, calls). The intersection count is shown in parentheses.

Cross-cell-line comparison of site counts

We use the Jaccard index to quantify how much the pooled, significance -filtered site sets overlap between every pair of cell lines. For each combination of FDR / rate-difference cutoff and minimum-replicate filter, sites passing both filters in each cell line are extracted and the pairwise Jaccard is computed:

\[J(A, B) = \frac{|A \cap B|}{|A \cup B|}\]

A high Jaccard index means the two cell lines call largely the same pooled rRNA positions, consistent with constitutively modified sites shared across cell lines. Each cell shows the Jaccard score and, in parentheses, the number of sites in the intersection. Rows = minimum n_samples; columns = FDR / rate-difference combination.

jaccard_pooled_rows <- do.call(rbind, lapply(cl_pairs_v_chisq, function(p) {
  do.call(rbind, lapply(seq_along(CUTOFF_COMBOS), function(i) {
    combo <- CUTOFF_COMBOS[[i]]
    do.call(rbind, lapply(NSAMP_CUTS_POOLED, function(n) {
      dt1 <- dat_list_pooled_chisq[[p[1]]]
      dt2 <- dat_list_pooled_chisq[[p[2]]]
      sites1 <- dt1$site[dt1$fdr < combo$fdr & dt1$rate_diff > combo$diff & dt1$n_samples >= n]
      sites2 <- dt2$site[dt2$fdr < combo$fdr & dt2$rate_diff > combo$diff & dt2$n_samples >= n]
      n_int  <- length(intersect(sites1, sites2))

                 combo_idx = i, combo_label = combo$label, min_nsamp = n,
                 jaccard = round(j, 3), n_intersect = n_int)

  }))
}))

Pearson’s r of %pU between cell lines

To assess whether the same sites are similarly modified across cell lines, we compute Pearson’s r on the pooled %pU values at sites passing the FDR / rate-difference filter and the minimum-replicate filter in both cell lines of each pair. High r across cell lines suggests that the detected modification pattern is consistent — i.e., the same rRNA sites are strongly modified regardless of cell line — which is expected for constitutively modified rRNA residues.

Sites that pass in one cell line but not the other are excluded from the correlation (they contribute to the Jaccard-style overlap question, not the quantitative agreement question).

pearson_pair_chisq <- function(dt1, dt2, combo, min_nsamp) {
  pass1 <- dt1[fdr < combo$fdr & rate_diff > combo$diff & n_samples >= min_nsamp,
               .(site, ratio1 = ratio)]
  pass2 <- dt2[fdr < combo$fdr & rate_diff > combo$diff & n_samples >= min_nsamp,
               .(site, ratio2 = ratio)]
  merged <- merge(pass1, pass2, by = "site")
  if (nrow(merged) < 3) return(data.frame(r = NA_real_, n_common = nrow(merged)))
  data.frame(
    r        = cor(merged$ratio1, merged$ratio2, method = "pearson"),
    n_common = nrow(merged)
  )
}
