Introduction

Quality control of the union cross-cell-line mRNA pU site calling across all 51 cell lines (with problematic replicates dropped based on BACS r < 0.3 in both reps).

The union strategy calls pU sites independently per cell line (FDR < 0.01, rate_diff > 0.05, depth ≥ 20, n_samples ≥ 1 if ≤2 replicates else ≥ 2), then takes the union across all 51 CLs. A site enters the global list if any cell line passes the per-CL threshold. The called_in field records which cell line(s) independently called each site. Cross-cell-line pooled stats are looked up from the existing joint all-51 run.

The primary quality stratification for union sites is n_called — the number of independent cell lines that called the site. All analyses below are tabbed by n_called threshold (≥1, ≥2, ≥3, ≥5, ≥10 CLs).

Motif analysis around pU sites

We extract ±10 bp flanking sequences around each union pU site from the reference genome, then compute position-specific nucleotide frequencies.

Known mRNA pseudouridine motifs:

  • TRUB1: GUΨCNANNC (genomic DNA: G at -2, T at -1, C at +1, A at +3, C at +6)
  • PUS7: UGΨAR (genomic DNA: T at -2, G at -1, A at +1, A/G at +2)

bed_file <- file.path(OUT_DIR, "union_mRNA.pseusite.bed")

bed <- fread(bed_file, header = FALSE, sep = "\t",

seqs  <- character(nrow(bed))
valid <- logical(nrow(bed))

for (i in seq_len(nrow(bed))) {
  chr    <- bed$chr[i]
  center <- bed$end[i]
  st     <- bed$strand[i]
  if (!chr %in% names(genome)) next
  chr_len <- length(genome[[chr]])
  left    <- center - flank
  right   <- center + flank
  if (left < 1 || right > chr_len) next
  s <- subseq(genome[[chr]], start = left, end = right)
  if (st == "-") s <- reverseComplement(s)
  seqs[i]  <- as.character(s)
  valid[i] <- TRUE
}

classify_pus <- function(mer9) {
  b <- strsplit(mer9, "")[[1]]
  if (b[1] == "T" && b[2] == "G" && b[4] == "A" && b[5] %in% c("A", "G")) return("PUS7")
  if (b[1] == "G" && b[2] == "T" && b[4] == "C" && b[6] == "A" && b[9] == "C") return("TRUB1")
  return("Other")
}
classes <- sapply(motif_9mer_char, classify_pus, USE.NAMES = FALSE)

for (cc in ncalled_cutoffs) {

make_class_heatmap <- function(class_seqs, class_name) {
  n <- length(class_seqs)
  if (n == 0) return(NULL)
  fm <- matrix(0, nrow = 4, ncol = seq_width,
               dimnames = list(base_order, positions))
  for (s in class_seqs) {
    bases <- strsplit(s, "")[[1]]
    for (j in seq_along(bases)) {
      b <- toupper(bases[j])
      if (b %in% base_order) fm[b, j] <- fm[b, j] + 1
    }
  }
  fp <- sweep(fm, 2, colSums(fm), "/") * 100
  pd <- melt(fp)
  colnames(pd) <- c("Base", "Position", "Frequency")
  pd$Position <- as.integer(as.character(pd$Position))
  pd$Base     <- factor(pd$Base, levels = base_order)
  ggplot(pd, aes(x = Position, y = Base, fill = Frequency)) +
    geom_tile(color = "white", linewidth = 0.3) +
    geom_text(aes(label = sprintf("%.0f", Frequency)), size = 2) +
    scale_fill_gradient2(low = "white", mid = "#ffffbf", high = "#d73027",
                         midpoint = 25, name = "% freq") +
    scale_x_continuous(breaks = positions) +
    geom_vline(xintercept = 0, color = "black", linewidth = 0.8, linetype = "dashed") +
    labs(title = sprintf("%s sites (n = %d)", class_name, n),
         x = "Position relative to pU site", y = NULL) +
    theme_bw(base_size = 9) +
    theme(panel.grid = element_blank())
}
